    <script>
        class HeroCarousel {
            constructor() {
                // Configuration and state
                this.currentIndex = 0;
                this.autoPlayInterval = null;
                this.autoPlayDelay = 6000;
                this.isDesktopView = window.matchMedia('(min-width: 768px)').matches;

                // DOM elements
                this.bgImages = document.querySelectorAll('.bg-image');
                this.mobileNavContainer = document.getElementById('mobile-nav');
                this.mobileNavWrapper = document.querySelector('#mobile-nav .swiper-wrapper');
                this.desktopNavContainer = document.getElementById('desktop-nav');
                this.ctaButton = document.getElementById('cta-button');
                this.bookingModal = document.getElementById('booking-modal');
                this.closeModalBtn = document.getElementById('close-modal-btn');
                
                // Swiper instance and thumbnail elements
                this.thumbnailSwiper = null;
                this.mobileThumbnails = null;
                this.desktopThumbnails = null;

                this.init();
            }

            init() {
                this.createNavigation();
                this.initSwiper();
                this.bindEvents();
                this.startAutoPlay();
                // Initial image source update
                this.updateImageSource();
                // Update active slide on load
                this.updateActiveSlide(this.currentIndex);
            }
            
            updateImageSource() {
                this.isDesktopView = window.matchMedia('(min-width: 768px)').matches;
                
                // Update main hero images
                this.bgImages.forEach(img => {
                    img.src = this.isDesktopView ? img.dataset.desktopSrc : img.dataset.mobileSrc;
                });

                // Update thumbnail images
                document.querySelectorAll('.thumbnail img, .desktop-thumbnail img').forEach(img => {
                    if (img.dataset.desktopSrc === "https://i.pinimg.com/736x/0c/06/77/0c06772d4c876f8bf3e302aac9ef329b.jpg") {
                        img.src = this.isDesktopView 
                                ? "https://placehold.co/200x112/1a1a1a/cccccc?text=Desktop_Thumb_1"
                                : "https://placehold.co/200x355/1a1a1a/cccccc?text=Mobile_Thumb_1";
                    } else {
                        img.src = this.isDesktopView ? img.dataset.desktopSrc.replace('1920x1080', '200x112') : img.dataset.mobileSrc.replace('1080x1920', '200x355');
                    }
                });
            }

            updateActiveSlide(index) {
                // Remove active classes from old slide
                this.bgImages[this.currentIndex].classList.remove('active');
                
                if (this.mobileThumbnails) {
                    this.mobileThumbnails[this.currentIndex].classList.remove('active');
                }
                
                if (this.desktopThumbnails) {
                    this.desktopThumbnails[this.currentIndex].classList.remove('active');
                }
                
                // Update the current index
                this.currentIndex = index;

                // Add active classes to the new slide
                this.bgImages[this.currentIndex].classList.add('active');
                if (this.mobileThumbnails) {
                    this.mobileThumbnails[this.currentIndex].classList.add('active');
                }
                if (this.desktopThumbnails) {
                    this.desktopThumbnails[this.currentIndex].classList.add('active');
                }
            }

            createNavigation() {
                // Clear any existing thumbnails before creating new ones
                this.mobileNavWrapper.innerHTML = '';
                this.desktopNavContainer.innerHTML = '';
                
                // Create mobile Swiper thumbnails and desktop side thumbnails
                this.bgImages.forEach((img, index) => {
                    // Mobile Thumbnail (for Swiper)
                    const swiperSlide = document.createElement('div');
                    swiperSlide.classList.add('swiper-slide');
                    const mobileThumbnailDiv = document.createElement('div');
                    mobileThumbnailDiv.classList.add('thumbnail');
                    const mobileThumbnailImg = document.createElement('img');
                    mobileThumbnailImg.dataset.mobileSrc = img.dataset.mobileSrc.replace('1080x1920', '200x355');
                    mobileThumbnailImg.dataset.desktopSrc = img.dataset.desktopSrc.replace('1920x1080', '200x112');
                    mobileThumbnailImg.alt = img.alt;
                    const progressBar = document.createElement('div');
                    progressBar.classList.add('progress-bar');
                    mobileThumbnailDiv.appendChild(mobileThumbnailImg);
                    mobileThumbnailDiv.appendChild(progressBar);
                    swiperSlide.appendChild(mobileThumbnailDiv);
                    this.mobileNavWrapper.appendChild(swiperSlide);

                    // Desktop Thumbnail (for side navigation)
                    const desktopThumbnailDiv = document.createElement('div');
                    desktopThumbnailDiv.classList.add('desktop-thumbnail');
                    const desktopThumbnailImg = document.createElement('img');
                    desktopThumbnailImg.dataset.mobileSrc = img.dataset.mobileSrc.replace('1080x1920', '200x355');
                    desktopThumbnailImg.dataset.desktopSrc = img.dataset.desktopSrc.replace('1920x1080', '200x112');
                    desktopThumbnailImg.alt = img.alt;
                    const desktopProgressBar = document.createElement('div');
                    desktopProgressBar.classList.add('progress-bar');
                    desktopThumbnailDiv.appendChild(desktopThumbnailImg);
                    desktopThumbnailDiv.appendChild(desktopProgressBar);
                    this.desktopNavContainer.appendChild(desktopThumbnailDiv);
                });

                // Re-select the created elements
                this.mobileThumbnails = document.querySelectorAll('#mobile-nav .thumbnail');
                this.desktopThumbnails = document.querySelectorAll('#desktop-nav .desktop-thumbnail');
            }
            
            initSwiper() {
                // Initialize Swiper only for mobile view
                this.thumbnailSwiper = new Swiper('.swiper-nav-thumbs', {
                    slidesPerView: 'auto',
                    spaceBetween: 12,
                    centeredSlides: false,
                    freeMode: true,
                    mousewheel: true,
                    pagination: {
                        el: '.swiper-pagination',
                        clickable: true,
                    },
                });
                
                this.thumbnailSwiper.on('slideChange', () => {
                    this.goToSlide(this.thumbnailSwiper.realIndex);
                });
            }

            bindEvents() {
                // Event listeners for mobile thumbnails
                this.mobileThumbnails.forEach((thumbnail, index) => {
                    thumbnail.addEventListener('click', () => this.goToSlide(index));
                });
                
                // Event listeners for desktop thumbnails
                this.desktopThumbnails.forEach((thumbnail, index) => {
                    thumbnail.addEventListener('click', () => this.goToSlide(index));
                });
                
                document.addEventListener('keydown', (e) => {
                    if (e.key === 'ArrowLeft') {
                        this.goToSlide(this.currentIndex - 1);
                    } else if (e.key === 'ArrowRight') {
                        this.goToSlide(this.currentIndex + 1);
                    }
                });

                window.addEventListener('resize', () => {
                    this.updateImageSource();
                });

                const container = document.querySelector('.hero-container');
                container.addEventListener('mouseenter', () => this.stopAutoPlay());
                container.addEventListener('mouseleave', () => this.startAutoPlay());

                this.ctaButton.addEventListener('click', () => {
                    this.bookingModal.showModal();
                });

                this.closeModalBtn.addEventListener('click', () => {
                    this.bookingModal.close();
                });
            }

            goToSlide(index) {
                if (index < 0) index = this.bgImages.length - 1;
                if (index >= this.bgImages.length) index = 0;
                
                if (index === this.currentIndex) return;

                this.updateActiveSlide(index);
                
                // Sync the swiper to the new slide only if it exists
                if (this.thumbnailSwiper) {
                    this.thumbnailSwiper.slideTo(index);
                }
                
                this.restartAutoPlay();
            }

            nextSlide() {
                this.goToSlide(this.currentIndex + 1);
            }

            startAutoPlay() {
                this.stopAutoPlay();
                this.autoPlayInterval = setInterval(() => {
                    this.nextSlide();
                }, this.autoPlayDelay);
            }
            
            stopAutoPlay() {
                if (this.autoPlayInterval) {
                    clearInterval(this.autoPlayInterval);
                    this.autoPlayInterval = null;
                }
            }
            
            restartAutoPlay() {
                this.stopAutoPlay();
                this.startAutoPlay();
            }
        }
